class FileHandler extends Handler {

    private static final long MAX_SIZE = 1024;
    private static final int MAX_FILES = 3;

    protected final String fileName;
    protected BufferedWriter writer;
    protected final int key;

    FileHandler(int key, String fileName) throws IOException {
        this.fileName = fileName;
        this.key = key;
        writer = new BufferedWriter(new FileWriter(fileName, true));
    }

    @Override
    void log(LogRecord record) throws IOException {

        if (record.key() != this.key) return; // 🔥 key filter

        rotateIfNeeded();

        writer.write(record.format());
        writer.newLine();
        writer.flush();
    }

    private void rotateIfNeeded() throws IOException {

        File file = new File(fileName);

        if (file.length() < MAX_SIZE) return;

        writer.close();

        // 🔥 TIMESTAMP rotation (ONLY ONE MOVE)

        Path rotated = Paths.get(fileName + "." + timestamp);

        Files.move(Paths.get(fileName), rotated);

        writer = new BufferedWriter(new FileWriter(fileName, true));

        // 🔥 cleanup old files
        cleanupOldFiles();
    }

    private void cleanupOldFiles() throws IOException {

        Path logDir = Paths.get("logs");

        if (!Files.exists(logDir)) return;

        Files.list(logDir)
                .filter(p -> p.getFileName().toString().startsWith(new File(fileName).getName() + "."))
                .sorted(Comparator.reverseOrder()) // newest first
                .skip(MAX_FILES)
                .forEach(p -> {
                    try {
                        Files.deleteIfExists(p);
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                });
    }
}

    private void rotateIfNeeded() throws IOException {

        File file = new File(fileName);

        if (file.length() < MAX_SIZE) return;

        writer.close();

        for (int i = MAX_FILES - 1; i >= 1; i--) {

            Path src = Paths.get(fileName + "." + i);
            Path destination = Paths.get(fileName + "." + (i + 1));

            if (Files.exists(src)) {

            }
        }

        Files.move(Paths.get(fileName),
                Paths.get(fileName + ".1"),
                StandardCopyOption.REPLACE_EXISTING);

    }

class LogDispatcher {

    private static final BlockingQueue<LogRecord> queue =
            new LinkedBlockingQueue<>(10000);

    private static final List<Handler> handlers = new ArrayList<>();

    static {
        startWorker();

        Runtime.getRuntime().addShutdownHook(new Thread(LogDispatcher::flush));
    }

    public static void addHandler(Handler handler) {
        handlers.add(handler);
    }

    public static void dispatch(LogRecord record) {

        if (!queue.offer(record)) {
            System.err.println("Log queue full: " + record.format());
        }
    }

    private static void startWorker() {

        Thread worker = new Thread(() -> {

            try {

                while (true) {

                    LogRecord record = queue.take();

                    for (Handler a : handlers) {
                        a.log(record);
                    }
                }

            } catch (Exception e) {
                e.printStackTrace();
            }

        });

        worker.setDaemon(true);
        worker.start();
    }

    public static void flush() {

        LogRecord record;

        while ((record = queue.poll()) != null) {

            for (Handler a : handlers) {

                try {
                    a.log(record);
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }
        }
    }
}
